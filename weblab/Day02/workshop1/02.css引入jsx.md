在 JSX 中引入 CSS，主要有以下几种方式：

## 1. 全局引入
在组件文件中直接 import CSS 文件，样式会全局生效：
```jsx
import './App.css';

function App() {
  return <div className="app">内容</div>;
}
```

## 2. CSS Modules
使用 CSS Modules 实现样式局部作用域：
```jsx
import styles from './App.module.css';

function App() {
  return <div className={styles.container}>内容</div>;
}
```
- 文件名格式：`[name].module.css`
- 类名会被编译为唯一哈希值
- 通过 `styles.className` 访问

## 3. 行内样式
使用 style 属性直接定义样式：
```jsx
function App() {
  const divStyle = {
    color: 'red',
    backgroundColor: 'blue',
    // 驼峰命名
  };

  return (
    <div style={divStyle}>
      {/* 或者直接写对象 */}
      <p style={{ fontSize: '16px', margin: 0 }}>内容</p>
    </div>
  );
}
```

## 4. CSS-in-JS
使用 styled-components 或 emotion 等库：
```jsx
import styled from 'styled-components';

const StyledDiv = styled.div`
  color: red;
  padding: 20px;
`;

function App() {
  return <StyledDiv>内容</StyledDiv>;
}
```

## 5. 条件类名
动态设置 className：
```jsx
function App({ isActive }) {
  const className = `base-class ${isActive ? 'active' : ''}`;
  
  // 或者使用 classnames 库
  const cn = require('classnames');
  const className = cn('base-class', { active: isActive });

  return <div className={className}>内容</div>;
}
```

## 关键知识点：
- **CSS Modules** 是最推荐的局部样式方案
- **全局引入** 要小心样式污染
- **行内样式** 优先级最高，但不利于维护
- **CSS-in-JS** 适合复杂动态样式
- 使用 `className` 而不是 `class`
